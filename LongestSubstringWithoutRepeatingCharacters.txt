3. Longest Substring Without Repeating Characters
Given a string s, find the length of the longest substring without repeating characters.

 

Example 1:

Input: s = "abcabcbb"
Output: 3
Explanation: The answer is "abc", with the length of 3.
-------------------------------------------------------
//aasshhik88
#include<iostream>
using namespace std;

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int heightlength = 0,sln = 0;
        string preString = "" ;
        for(int i=0; i<s.length(); i++){
            if(i==0){
              preString +=s[i]; 
            }else{
                int prelength = 0;
                for(int j =0; j<preString.length(); j++){
                    if(preString[j] == s[i]){
                        prelength = 1;
                        sln = j;
                        break;
                    }
                }
                if(prelength == 1){
                    if(preString.length() > heightlength){
                        heightlength = preString.length();
                    }
                    string temp="";
                    for(int j = sln+1; j<preString.length(); j++){
                        temp += preString[j];
                    }
                    
                    preString = temp;
                    preString +=s[i];
                }else{
                    preString +=s[i];
                }
            }
            
        }
        
        if(preString.length() > heightlength){
            heightlength = preString.length();
        }
        
        return heightlength;
    }
};
